# P4刷写C6固件方案



## C6固件准备

### 在c6的从机组件中编译烧录文件

这里如果固件准备好可直接跳过

需要有esp_hosted这个组件

不需要接COM口运行（建议在单独的文件运行）

```bash

idf.py -C managed_components/espressif__esp_hosted/slave/ -B build_slave build flash monitor -p COM3 # ESP32C6连接的COM口
```

运行完之后，虽然没烧录进去，但是已经编译好了C6的固件

可以在build_slave文件夹里的flasher_args.json文件看到我们需要烧录的bin文件信息以及他们需要烧录的地址

### 将C6固件写入P4中

想要P4给C6进行烧录，需要P4中存有C6的固件，这里有两种方式

1. 把 C6 的 bin 打进 P4 固件镜像里（编译时嵌入）
2. 用 esptool 把 C6 的 bin 写到 P4 的数据分区（运行时读取）

对比两个方案的优缺点

1. 固件版本基本固定，很少换
   选编译时嵌入更简单，代码最少，稳定性高。
   缺点是每次换 C6 固件，都要重编并重烧 P4 主程序。
2. C6 固件经常更新，想减少 P4 主程序改动
   选写入 P4 数据分区更好。
   优点是只更新数据分区，不动 P4 主应用，产线维护更轻。
   缺点是要做一层包管理：镜像清单、地址、长度、CRC 校验。

这里选择第二种，相对独立，好维护。



#### 为固件划好分区

```csv
# Name,   Type, SubType, Offset,  Size
nvs,      data, nvs,     0x9000,  0x6000,
phy_init, data, phy,     0xf000,  0x1000,
ota_data, data, ota,     0x10000, 0x2000,
ota_0,    app,  ota_0,   0x20000, 0x300000,  
ota_1,    app,  ota_1,   0x320000,0x300000,   
factory,  app,  factory, 0x620000,0x300000,   
storage,  data, spiffs,  0x920000,0x200000,
c6_fw_pkg, data, 0x41, 0xB20000, 0x200000,   
multi_model,    data, 0x40,     0xF00000,   0x20000
gait_model, data, 0x40,     0xF20000,   0x50000
```

这里取storage后面2MB的大小给C6固件（看了C6固件总大小大概1.2MB左右，可以留点余量）

<img src="P4%E5%88%B7%E5%86%99C6%E5%9B%BA%E4%BB%B6%E6%96%B9%E6%A1%88.assets/image-20260409161727349.png" alt="image-20260409161727349" style="zoom:80%;" />

实际总量约 1151KB，约 1.12MB。放进 2MB 分区没问题。

可以使用如下地址刷写给p4

1. 分区基址：0xB20000，大小 0x200000（来自 [partitions.csv](vscode-file://vscode-app/d:/Program Files/Microsoft VS Code/41dd792b5e/resources/app/out/vs/code/electron-browser/workbench/workbench.html)）
2. bootloader 存储地址：0xB20000，预留 0x10000
3. partition-table 存储地址：0xB30000，预留 0x1000
4. otadata 存储地址：0xB31000，预留 0x2000
5. app 存储地址：0xB40000，预留到分区末尾（约 0x1E0000 可用）

确认好刷写地址之后，使用esptool写入



#### 刷C6固件进入P4

打开idf终端

检查文件是否在目录中

<img src="P4%E5%88%B7%E5%86%99C6%E5%9B%BA%E4%BB%B6%E6%96%B9%E6%A1%88.assets/image-20260409162707499.png" alt="image-20260409162707499" style="zoom:67%;" />

COMx替换为自己的端口号

```shell
# 擦除
esptool.py --chip esp32p4 --port COM5 --baud 921600 erase_region 0xB20000 0x200000
# 刷写
esptool.py --chip esp32p4 --port COM5 --baud 921600 write_flash --force 0xB20000 c6_fw_bin/bootloader.bin 0xB30000 c6_fw_bin/partition-table.bin 0xB31000 c6_fw_bin/ota_data_initial.bin 0xB40000 c6_fw_bin/network_adapter.bin
```

### `c6_flasher_sdio.c`

```c
#include "c6_flasher_sdio.h"

#include <string.h>
#include <stdlib.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_log.h"
#include "esp_system.h"
#include "driver/uart.h"
#include "driver/sdmmc_host.h"
#include "esp_loader.h"
#include "esp_loader_io.h"
#include "esp32_sdio_port.h"
#include "esp_wifi_remote.h"
#include "esp_hosted.h"

#include "esp_partition.h"
#include "config.h"

#define TAG "c6_flasher_sdio"

// 是否打印C6中的日志，调试使用
#define C6_SDIO_PRINTF_SWITCH true

#define FLASH_BLOCK_SIZE 1024
#define PROGRESS_LOG_PERIOD_MS 5000

// 需要用真实长度替换，命令查看：Get-ChildItem .\c6_fw_bin\*.bin | Select-Object Name,Length
#define C6_BOOT_SIZE (24752)
#define C6_PART_SIZE (3072)
#define C6_OTA_SIZE (8192)
#define C6_APP_SIZE (1141072)

// 在 c6_fw_pkg 分区中的偏移
#define PKG_OFF_BOOT 0x00000
#define PKG_OFF_PART 0x10000
#define PKG_OFF_OTA 0x11000
#define PKG_OFF_APP 0x20000

// C6 目标烧写地址（来自 C6 分区表）
#define C6_ADDR_BOOT 0x0000
#define C6_ADDR_PART 0x8000
#define C6_ADDR_OTA 0xD000
#define C6_ADDR_APP 0x10000

// P4 -> C6 SDIO 引脚（参考 esp32_sdio_example）
#define CONFIG_C6_SDIO_SLOT SDMMC_HOST_SLOT_1
#define CONFIG_C6_SDIO_BOOT_PIN 32
#define CONFIG_C6_SDIO_D0_PIN 14
#define CONFIG_C6_SDIO_D1_PIN 15
#define CONFIG_C6_SDIO_D2_PIN 16
#define CONFIG_C6_SDIO_D3_PIN 17
#define CONFIG_C6_SDIO_CLK_PIN 18
#define CONFIG_C6_SDIO_CMD_PIN 19

typedef struct
{
    const char *name;
    uint32_t pkg_offset;
    size_t size;
    uint32_t target_addr;
} c6_image_t;

#define BUF_LEN 128

static uint8_t s_buf[BUF_LEN] = {0};

static void c6_sdio_slave_monitor(void *arg)
{
    (void)arg;

    // Initialize UART
    uart_config_t uart_config = {
        .baud_rate = 115200,
        .data_bits = UART_DATA_8_BITS,
        .parity = UART_PARITY_DISABLE,
        .stop_bits = UART_STOP_BITS_1,
        .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 0)
        .source_clk = UART_SCLK_DEFAULT,
#endif
    };

    ESP_ERROR_CHECK(uart_param_config(CONFIG_C6_UART_NUM, &uart_config));

    ESP_ERROR_CHECK(uart_set_pin(CONFIG_C6_UART_NUM, CONFIG_C6_UART_TX_PIN, CONFIG_C6_UART_RX_PIN,
                                 UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE));

    ESP_ERROR_CHECK(uart_driver_install(CONFIG_C6_UART_NUM, BUF_LEN * 4, BUF_LEN * 4, 0, NULL, 0));
    while (1)
    {
        int rx_bytes = uart_read_bytes(CONFIG_C6_UART_NUM, s_buf, BUF_LEN - 1, 100 / portTICK_PERIOD_MS);
        if (rx_bytes > 0)
        {
            s_buf[rx_bytes] = '\0';
            printf("%s", s_buf);
        }
        else if (rx_bytes < 0)
        {
            ESP_LOGW(TAG, "uart_read_bytes failed on monitor uart");
            vTaskDelay(pdMS_TO_TICKS(200));
        }
    }
}

static esp_loader_error_t connect_to_target_sdio(void)
{
    esp_loader_connect_args_t connect_config = ESP_LOADER_CONNECT_DEFAULT();
    esp_loader_error_t err = esp_loader_connect(&connect_config);
    if (err != ESP_LOADER_SUCCESS)
    {
        ESP_LOGE(TAG, "Cannot connect to target by SDIO, err=%d", err);
        return err;
    }
    return ESP_LOADER_SUCCESS;
}

static esp_err_t flash_image_from_pkg(const esp_partition_t *pkg, const c6_image_t *img)
{
    uint8_t payload[FLASH_BLOCK_SIZE];
    size_t aligned_size = (img->size + 3U) & ~3U;
    size_t sent = 0;
    TickType_t last_log_tick = xTaskGetTickCount();

    if ((img->pkg_offset + img->size) > pkg->size)
    {
        ESP_LOGE(TAG, "%s out of c6_fw_pkg partition range", img->name);
        return ESP_ERR_INVALID_SIZE;
    }

    esp_loader_error_t lerr = esp_loader_flash_start(img->target_addr, aligned_size, FLASH_BLOCK_SIZE);
    if (lerr != ESP_LOADER_SUCCESS)
    {
        ESP_LOGE(TAG, "flash_start failed for %s, err=%d", img->name, lerr);
        return ESP_FAIL;
    }

    while (sent < aligned_size)
    {
        size_t to_send = (aligned_size - sent > FLASH_BLOCK_SIZE) ? FLASH_BLOCK_SIZE : (aligned_size - sent);
        size_t real_left = (sent < img->size) ? (img->size - sent) : 0;
        size_t real_read = (real_left > to_send) ? to_send : real_left;

        if (real_read > 0)
        {
            esp_err_t err = esp_partition_read(pkg, img->pkg_offset + sent, payload, real_read);
            if (err != ESP_OK)
            {
                ESP_LOGE(TAG, "Partition read failed for %s at +0x%lx, err=0x%x",
                         img->name, (unsigned long)sent, err);
                return err;
            }
        }

        if (real_read < to_send)
        {
            memset(payload + real_read, 0xFF, to_send - real_read);
        }

        lerr = esp_loader_flash_write(payload, to_send);
        if (lerr != ESP_LOADER_SUCCESS)
        {
            ESP_LOGE(TAG, "flash_write failed for %s at +0x%lx, err=%d",
                     img->name, (unsigned long)sent, lerr);
            return ESP_FAIL;
        }

        sent += to_send;

        TickType_t now_tick = xTaskGetTickCount();
        if ((now_tick - last_log_tick) >= pdMS_TO_TICKS(PROGRESS_LOG_PERIOD_MS))
        {
            uint32_t percent = (uint32_t)((sent * 100U) / aligned_size);
            ESP_LOGI(TAG, "%s progress: %lu%% (%lu/%lu)",
                     img->name,
                     (unsigned long)percent,
                     (unsigned long)sent,
                     (unsigned long)aligned_size);
            last_log_tick = now_tick;
        }
    }

    ESP_LOGI(TAG, "%s progress: 100%% (%lu/%lu)",
             img->name,
             (unsigned long)aligned_size,
             (unsigned long)aligned_size);
    ESP_LOGI(TAG, "%s flashed done", img->name);
    return ESP_OK;
}

static void stop_hosted_and_release_sdio(void)
{
    esp_err_t err;
    sdmmc_host_state_t state = {0};

    err = esp_wifi_remote_stop();
    if (err != ESP_OK)
    {
        ESP_LOGW(TAG, "esp_wifi_remote_stop ret=0x%x", err);
    }

    err = esp_wifi_remote_deinit();
    if (err != ESP_OK)
    {
        ESP_LOGW(TAG, "esp_wifi_remote_deinit ret=0x%x", err);
    }

    int hosted_ret = esp_hosted_deinit();
    if (hosted_ret != ESP_OK)
    {
        ESP_LOGW(TAG, "esp_hosted_deinit ret=%d", hosted_ret);
    }

    vTaskDelay(pdMS_TO_TICKS(100));

    err = sdmmc_host_get_state(&state);
    if (err == ESP_OK && state.host_initialized)
    {
        err = sdmmc_host_deinit();
        if (err != ESP_OK)
        {
            ESP_LOGW(TAG, "sdmmc_host_deinit ret=0x%x", err);
        }
    }
}

void c6_flasher_sdio_task(void *arg)
{
    (void)arg;
    ESP_LOGI(TAG, "C6 SDIO flasher task started");

    // 先停 hosted 上层，再释放 SDMMC host，避免竞争后又被重新拉起
    stop_hosted_and_release_sdio();

    const loader_esp32_sdio_config_t config = {
        .slot = CONFIG_C6_SDIO_SLOT,
        .max_freq_khz = SDMMC_FREQ_DEFAULT,
        .reset_trigger_pin = CONFIG_C6_RESET_TRIGGER_PIN,
        .boot_pin = CONFIG_C6_SDIO_BOOT_PIN,
        .bus_width = SDIO_4BIT,
        .sdio_d0_pin = CONFIG_C6_SDIO_D0_PIN,
        .sdio_d1_pin = CONFIG_C6_SDIO_D1_PIN,
        .sdio_d2_pin = CONFIG_C6_SDIO_D2_PIN,
        .sdio_d3_pin = CONFIG_C6_SDIO_D3_PIN,
        .sdio_clk_pin = CONFIG_C6_SDIO_CLK_PIN,
        .sdio_cmd_pin = CONFIG_C6_SDIO_CMD_PIN,
    };

    if (loader_port_esp32_sdio_init(&config) != ESP_LOADER_SUCCESS)
    {
        ESP_LOGE(TAG, "SDIO initialization failed");
        vTaskDelete(NULL);
        return;
    }

    if (connect_to_target_sdio() == ESP_LOADER_SUCCESS)
    {
        esp_err_t err;
        const esp_partition_t *pkg = esp_partition_find_first(ESP_PARTITION_TYPE_DATA, 0x41, "c6_fw_pkg");
        const c6_image_t images[] = {
            {"bootloader", PKG_OFF_BOOT, C6_BOOT_SIZE, C6_ADDR_BOOT},
            {"partition", PKG_OFF_PART, C6_PART_SIZE, C6_ADDR_PART},
            {"otadata", PKG_OFF_OTA, C6_OTA_SIZE, C6_ADDR_OTA},
            {"app", PKG_OFF_APP, C6_APP_SIZE, C6_ADDR_APP},
        };

        if (pkg == NULL)
        {
            ESP_LOGE(TAG, "Partition c6_fw_pkg not found");
            loader_port_esp32_sdio_deinit();
            vTaskDelete(NULL);
            return;
        }

        for (size_t i = 0; i < (sizeof(images) / sizeof(images[0])); ++i)
        {
            err = flash_image_from_pkg(pkg, &images[i]);
            if (err != ESP_OK)
            {
                ESP_LOGE(TAG, "Flashing failed at image %s", images[i].name);
                loader_port_esp32_sdio_deinit();
                vTaskDelete(NULL);
                return;
            }
        }

        esp_loader_reset_target();
        loader_port_esp32_sdio_deinit();
        vTaskDelay(100 / portTICK_PERIOD_MS);

        ESP_LOGI(TAG, "C6 SDIO flasher task completed");

#if (C6_SDIO_PRINTF_SWITCH == true)
        ESP_LOGI(TAG, "********************************************");
        ESP_LOGI(TAG, "*** Logs below are print from slave .... ***");
        ESP_LOGI(TAG, "********************************************");
        xTaskCreate(c6_sdio_slave_monitor, "c6_sdio_monitor", 2048, NULL, configMAX_PRIORITIES - 1, NULL);
#endif

        ESP_LOGI(TAG, "P4 will restart in 5 seconds...");
        vTaskDelay(pdMS_TO_TICKS(5000));
        esp_restart();
    }
    else
    {
        ESP_LOGE(TAG, "Connect to C6 by SDIO failed");
        loader_port_esp32_sdio_deinit();
    }

    vTaskDelete(NULL);
}
```
